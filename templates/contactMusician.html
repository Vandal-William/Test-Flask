Groupe: 1 Lister le nom et le prénom des musiciens, ordonnées par leur date de naissance

SELECT first_name, last_name
FROM artist
ORDER BY birthdate;

Groupe: 2 Lister les musiciens nés entre 1970 et 1990

SELECT last_name, first_name, birthdate
FROM artist
WHERE birthdate
BETWEEN '1970-01-01' AND '1990-12-31';

Groupe: 4 Lister les groupes et leur nombre de membres.

SELECT g.name AS groupe, COUNT(ag.artist_id) AS nombre_membres
FROM `group` AS g
INNER JOIN artist_group AS ag ON g.id = ag.group_id
GROUP BY g.id;
